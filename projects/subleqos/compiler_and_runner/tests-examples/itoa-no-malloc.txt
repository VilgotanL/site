char* _itoa_with_buff(int i, int* arr) {
    if(i == 0) return "0";
    int negative = i < 0;
    if(negative) i = -i;
    int* ptr = arr+len(arr)-1;
    int* start;
    
    int mult = 1;
    while(mult <= i) {
        *ptr = mult;
        ptr--;
        mult = mult+mult+mult+mult+mult;
        mult = mult+mult;
        if(ptr <= arr+1) return "(big)";
    }
    if(negative) { *ptr = '-'; start = ptr; }
    else { start = ptr+1; }
    ptr++;
    
    while(ptr < arr+len(arr)) {
        int digit = 0;
        while(i >= 0) {
            i -= *ptr;
            digit++;
        }
        i += *ptr;
        digit--;
        *ptr = '0'+digit;
        ptr++;
    }
    *(start-1) = arr+len(arr)-start;
    return start;
}
